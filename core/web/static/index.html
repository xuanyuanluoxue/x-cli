当前可用的 API（用 curl 测试）

# 健康检查（无需 token）
curl http://127.0.0.1:8421/api/health

# 列出任务（需要 X-Web-Token header）
curl -H "X-Web-Token: YOUR_TOKEN" http://127.0.0.1:8421/api/tasks

# 创建任务
curl -X POST -H "X-Web-Token: YOUR_TOKEN" -H "Content-Type: application/json" \
  -d '{"name":"测试任务","priority":"high"}' \
  http://127.0.0.1:8421/api/tasks
